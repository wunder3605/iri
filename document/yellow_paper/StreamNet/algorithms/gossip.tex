\begin{algorithm}
\SetKwData{Left}{left}\SetKwData{This}{this}\SetKwData{Up}{up}
\SetKwFunction{Union}{Union}\SetKwFunction{FindCompress}{FindCompress}
\SetKwInOut{Input}{input}\SetKwInOut{Output}{output}

\KwIn{ Graph $G=<B, g, P, E>$ }

    \While {Node is running}{
        \uIf{Generate block $b$}{
            Broadcast $b$ to neighbors \;
        } 
        \uIf{Receive block $b$}{
            $h \gets Hash(b)$ \;
            $cache[h] \gets b$ \;
            Broadcast $h$ to neighbors \;
        } 
        \uIf{Received request $h$ from neighbor $n$ }{
            $b \gets cache[h]$ \;
            Send $b$ to $n$ \;
        } 
        \uIf{Received hash $h$ from neighbor $n$}{
            $b \gets cache[h]$ \;
            \uIf{ $b = NULL $ } {  
                Send request $h$ to $n$ \;
            }
        } 
    }

\caption{{ The Direct Signal Gossip Algorithm.}}
\label{algo:gossip}
\end{algorithm}
