\begin{algorithm}
\SetKwData{Left}{left}\SetKwData{This}{this}\SetKwData{Up}{up}

\KwIn{ The local state $G$ = $<B,g,P,E>$  and a tip block $b \in B$ }
\KwOut{ The block list of total top order starting from Genesis block to the giving block $b$ in $G$ }

\Do { $b$ = $g$} {
  $p$ $\longleftarrow$ Parent($G,b$)  \;
  $subTopOrder$ $\longleftarrow$ $\perp$ \;
  $B_\Delta$ $\longleftarrow$ Past($G$,$b$) - Past($G$,$p$) \;
  \Do { $B_\Delta$ $\neq$ 0 } {
    $G'$ $\longleftarrow$ $<B_\Delta,g,P,E>$ \;
    $B'_\Delta$ $\longleftarrow$ \{x $||$ Before($G'$,$x$) = 0\} \;
    \text{Sort all blocks in $B'_\Delta$ in order as $b'_1,b'_2,...,b'_k$} \\
      \text{such that $\forall$1$\leq i \leq j \leq k$, Hash($b'_i$) $\leq$ Hash($b'_j$)} \;  
    $subTopOrder$ $\longleftarrow$ $list + b'_1 + b'_2 + ... + b'_k$ \;
    $B_\Delta$ $\longleftarrow$ $B_\Delta$ - $B'_\Delta$ \;
  }
  $L$ = $subTopOrder$ + $L$ \;
  $b$ = $p$ \;
}


\Return{$L$} \;

\caption{{\sc The Definition of ConfluxOrder($G$, $b$).}}
\label{algo:conflux_order}
\end{algorithm}
